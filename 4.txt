<script>
  let iterations = [];
  let currentIndex = -1;
  let lastGeneratedJSON = null;
  let ws = null;

  // -----------------------------
  // URL PARAMETER INGESTION + AUTO-ADVANCE
  // -----------------------------
  function ingestURLParamsAndAutoAdvance() {
    const params = new URLSearchParams(window.location.search);
    if (params.toString().length === 0) return;

    let foundAny = false;

    const mapping = {
      domain: 'domain',
      postAddress: 'postAddress',
      sense: 'sense',
      triage: 'triage',
      explain: 'explain',
      adapt: 'adapt'
    };

    for (const key in mapping) {
      if (params.has(key)) {
        document.getElementById(mapping[key]).value = params.get(key);
        foundAny = true;
      }
    }

    if (params.has('payload')) {
      try {
        const data = JSON.parse(params.get('payload'));
        for (const key in data) {
          if (document.getElementById(key)) {
            document.getElementById(key).value = data[key];
          }
        }
        foundAny = true;
      } catch (e) {
        console.warn("Invalid JSON payload:", e);
      }
    }

    if (foundAny) autoCreateIteration();
  }

  function autoCreateIteration() {
    const record = buildRecord();
    iterations.push(record);
    currentIndex = iterations.length - 1;
    clearFields();
    renderTimeline();
    renderOutput();
  }

  // -----------------------------
  // JSON INJECTION
  // -----------------------------
  function loadFromJSON() {
    try {
      const data = JSON.parse(document.getElementById('injectBox').value);
      for (const key in data) {
        if (document.getElementById(key)) {
          document.getElementById(key).value = data[key];
        }
      }
      alert("Fields populated from JSON input.");
    } catch (e) {
      alert("Invalid JSON.");
    }
  }

  // -----------------------------
  // RECORD BUILDING
  // -----------------------------
  function buildRecord() {
    return {
      iteration: iterations.length + 1,
      timestamp: new Date().toISOString(),
      postAddress: document.getElementById('postAddress').value,
      domain: document.getElementById('domain').value,
      sense: document.getElementById('sense').value,
      triage: document.getElementById('triage').value,
      explain: document.getElementById('explain').value,
      adapt: document.getElementById('adapt').value
    };
  }

  function clearFields() {
    document.getElementById('sense').value = "";
    document.getElementById('triage').value = "";
    document.getElementById('explain').value = "";
    document.getElementById('adapt').value = "";
  }

  // -----------------------------
  // TWO-WAY LLM INTEGRATION
  // -----------------------------
  async function sendToLLM() {
    const url = document.getElementById('llmUrl').value.trim();
    if (!url) {
      alert("Please enter an LLM endpoint URL.");
      return;
    }

    const payload = buildRecord();

    try {
      const response = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      });

      const data = await response.json();

      // Populate fields with LLM response
      for (const key in data) {
        if (document.getElementById(key)) {
          document.getElementById(key).value = data[key];
        }
      }

      // Auto-create new iteration from LLM output
      autoCreateIteration();

    } catch (e) {
      alert("LLM request failed: " + e);
    }
  }

  // -----------------------------
  // JSON GENERATION + POST SIM
  // -----------------------------
  function generateJSON() {
    lastGeneratedJSON = buildRecord();
    document.getElementById('output').textContent =
      JSON.stringify(lastGeneratedJSON, null, 2);
  }

  function simulatePOST() {
    if (!lastGeneratedJSON) generateJSON();
    alert("Simulated POST:\n\n" + JSON.stringify(lastGeneratedJSON, null, 2));
  }

  // -----------------------------
  // ITERATION MANAGEMENT
  // -----------------------------
  function saveIteration() {
    const record = buildRecord();
    iterations.push(record);
    currentIndex = iterations.length - 1;
    clearFields();
    renderTimeline();
    renderOutput();
  }

  function renderTimeline() {
    const list = document.getElementById('timelineList');
    list.innerHTML = "";

    iterations.forEach((it, idx) => {
      const div = document.createElement('div');
      div.className = "timeline-item" + (idx === currentIndex ? " active" : "");
      div.textContent = "Iteration " + it.iteration + " — " + it.timestamp;
      div.onclick = () => loadIteration(idx);
      list.appendChild(div);
    });
  }

  function loadIteration(index) {
    if (index < 0 || index >= iterations.length) return;

    currentIndex = index;
    const it = iterations[index];

    document.getElementById('domain').value = it.domain;
    document.getElementById('postAddress').value = it.postAddress;
    document.getElementById('sense').value = it.sense;
    document.getElementById('triage').value = it.triage;
    document.getElementById('explain').value = it.explain;
    document.getElementById('adapt').value = it.adapt;

    renderTimeline();
    renderOutput();
  }

  function prevIteration() {
    if (currentIndex > 0) loadIteration(currentIndex - 1);
  }

  function nextIteration() {
    if (currentIndex < iterations.length - 1) loadIteration(currentIndex + 1);
  }

  function renderOutput() {
    document.getElementById('output').textContent =
      JSON.stringify(iterations, null, 2);
  }

  // -----------------------------
  // WEBSOCKET LIVE INGESTION
  // -----------------------------
  function connectWebSocket() {
    const url = document.getElementById('wsUrl').value.trim();
    if (!url) {
      alert("Please enter a WebSocket URL.");
      return;
    }

    if (ws && ws.readyState === WebSocket.OPEN) {
      ws.close();
    }

    ws = new WebSocket(url);
    const statusEl = document.getElementById('wsStatus');

    ws.onopen = () => statusEl.textContent = "(WebSocket: connected)";
    ws.onclose = () => statusEl.textContent = "(WebSocket: disconnected)";
    ws.onerror = () => statusEl.textContent = "(WebSocket: error)";

    ws.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data);
        for (const key in data) {
          if (document.getElementById(key)) {
            document.getElementById(key).value = data[key];
          }
        }
        autoCreateIteration();
      } catch (e) {
        console.warn("Invalid JSON from WebSocket:", e);
      }
    };
  }

  // -----------------------------
  // INITIALIZATION
  // -----------------------------
  ingestURLParamsAndAutoAdvance();
</script>

</body>
</html>
