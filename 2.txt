# **AOS Workspace with Iteration Timeline Navigation**

Save this as `aos_workspace_timeline.html` and open it in any browser.

It includes:

- Four notepad‑style sections  
- Save + Next Iteration  
- A timeline panel  
- Previous / Next navigation  
- Click‑to‑load iteration selection  

---

```html
<!DOCTYPE html>
<html>
<head>
<meta charset="UTF-8">
<title>AOS Iterative Workspace with Timeline</title>
<style>
  body { font-family: sans-serif; margin: 20px; display: flex; gap: 30px; }
  #main { flex: 3; }
  #timeline { flex: 1; border-left: 2px solid #ccc; padding-left: 20px; }

  h1 { margin-bottom: 10px; }
  h2 { margin-top: 25px; }
  textarea {
    width: 100%;
    height: 140px;
    font-family: monospace;
    font-size: 14px;
    padding: 8px;
    margin-top: 8px;
  }

  .meta input {
    padding: 6px;

    width: 250px;
  }

    padding: 10px 20px;

    margin-right: 10px;
  }

  #output {
    white-space: pre;
    background: #f5f5f5;
    padding: 10px;
    margin-top: 20px;
    border: 1px solid #ccc;
  }

  .timeline-item {
    padding: 8px;
    margin-bottom: 8px;
    border: 1px solid #ddd;
    cursor: pointer;
    border-radius: 4px;
  }
  .timeline-item.active {
    background: #e0f0ff;
    border-color: #3399ff;
  }
</style>
</head>
<body>

<div id="main">
  <h1>AOS Iterative Analysis Workspace</h1>

  <div class="meta">
    <label>Domain:
      <input id="domain" placeholder="financial / law_enforcement / system_logs / social">
    </label>
  </div>

  <h2>1. Sense</h2>
  <textarea id="sense"></textarea>

  <h2>2. Triage</h2>
  <textarea id="triage"></textarea>

  <h2>3. Explain</h2>
  <textarea id="explain"></textarea>

  <h2>4. Adapt</h2>
  <textarea id="adapt"></textarea>

  <button onclick="saveIteration()">Save + Next Iteration</button>
  <button onclick="prevIteration()">Previous</button>
  <button onclick="nextIteration()">Next</button>

  <h2>Saved Iterations (JSON)</h2>
  <div id="output"></div>
</div>

<div id="timeline">
  <h2>Timeline</h2>
  <div id="timelineList"></div>
</div>

<script>
  let iterations = [];
  let currentIndex = -1;

  function renderTimeline() {
    const list = document.getElementById('timelineList');
    list.innerHTML = "";

    iterations.forEach((it, idx) => {
      const div = document.createElement('div');
      div.className = "timeline-item" + (idx === currentIndex ? " active" : "");
      div.textContent = "Iteration " + it.iteration + " — " + it.timestamp;
      div.onclick = () => loadIteration(idx);
      list.appendChild(div);
    });
  }

  function loadIteration(index) {
    if (index < 0 || index >= iterations.length) return;

    currentIndex = index;
    const it = iterations[index];

    document.getElementById('domain').value = it.domain;
    document.getElementById('sense').value = it.sense;
    document.getElementById('triage').value = it.triage;
    document.getElementById('explain').value = it.explain;
    document.getElementById('adapt').value = it.adapt;

    renderTimeline();
    renderOutput();
  }

  function saveIteration() {
    const record = {
      iteration: iterations.length + 1,
      timestamp: new Date().toISOString(),
      domain: document.getElementById('domain').value,
      sense: document.getElementById('sense').value,
      triage: document.getElementById('triage').value,
      explain: document.getElementById('explain').value,
      adapt: document.getElementById('adapt').value
    };

    iterations.push(record);
    currentIndex = iterations.length - 1;

    // Clear fields for next iteration
    document.getElementById('sense').value = "";
    document.getElementById('triage').value = "";
    document.getElementById('explain').value = "";
    document.getElementById('adapt').value = "";

    renderTimeline();
    renderOutput();
  }

  function prevIteration() {
    if (currentIndex > 0) {
      loadIteration(currentIndex - 1);
    }
  }

  function nextIteration() {
    if (currentIndex < iterations.length - 1) {
      loadIteration(currentIndex + 1);
    }
  }

  function renderOutput() {
    document.getElementById('output').textContent =
      JSON.stringify(iterations, null, 2);
  }
</script>

</body>
</html>
```
