<script>
  let iterations = [];
  let currentIndex = -1;

  function buildRecord() {
    return {
      iteration: iterations.length + 1,
      timestamp: new Date().toISOString(),
      domain: document.getElementById('domain').value,
      sense: document.getElementById('sense').value,
      triage: document.getElementById('triage').value,
      explain: document.getElementById('explain').value,
      adapt: document.getElementById('adapt').value
    };
  }

  function clearFields() {
    document.getElementById('sense').value = "";
    document.getElementById('triage').value = "";
    document.getElementById('explain').value = "";
    document.getElementById('adapt').value = "";
  }

  function saveIteration() {
    const record = buildRecord();
    iterations.push(record);
    currentIndex = iterations.length - 1;
    clearFields();
    renderOutput();
    renderLogs();
  }

  function generateJSON() {
    const record = buildRecord();
    document.getElementById('output').textContent =
      JSON.stringify(record, null, 2);
  }

  function loadFromJSON() {
    try {
      const data = JSON.parse(document.getElementById('injectBox').value);
      for (const key in data) {
        if (document.getElementById(key)) {
          document.getElementById(key).value = data[key];
        }
      }
      alert("Fields populated from JSON input.");
    } catch (e) {
      alert("Invalid JSON.");
    }
  }

  function renderOutput() {
    document.getElementById('output').textContent =
      JSON.stringify(iterations, null, 2);
  }

  function renderLogs() {
    const list = document.getElementById('logList');
    list.innerHTML = "";

    iterations.forEach(it => {
      const div = document.createElement('div');
      div.className = "log-entry";

      div.innerHTML = `
        <h3>Iteration ${it.iteration} — ${it.timestamp}</h3>
        <div class="log-field"><span>Domain:</span> ${it.domain}</div>
        <div class="log-field"><span>Sense:</span> ${it.sense}</div>
        <div class="log-field"><span>Triage:</span> ${it.triage}</div>
        <div class="log-field"><span>Explain:</span> ${it.explain}</div>
        <div class="log-field"><span>Adapt:</span> ${it.adapt}</div>
      `;

      list.appendChild(div);
    });
  }

  async function sendToLLM() {
    const url = document.getElementById('llmUrl').value.trim();
    if (!url) {
      alert("Please enter an LLM endpoint URL.");
      return;
    }

    const payload = buildRecord();

    try {
      const response = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      });

      const data = await response.json();

      for (const key in data) {
        if (document.getElementById(key)) {
          document.getElementById(key).value = data[key];
        }
      }

      saveIteration();

    } catch (e) {
      alert("LLM request failed: " + e);
    }
  }

  async function runLLMLoop() {
    const url = document.getElementById('llmUrl').value.trim();
    const count = parseInt(document.getElementById('loopCount').value, 10) || 1;

    if (!url) {
      alert("Please enter an LLM endpoint URL.");
      return;
    }

    let payload = buildRecord();

    for (let i = 0; i < count; i++) {
      try {
        const response = await fetch(url, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(payload)
        });

        const data = await response.json();

        for (const key in data) {
          if (document.getElementById(key)) {
            document.getElementById(key).value = data[key];
          }
        }

        saveIteration();
        payload = iterations[iterations.length - 1];

      } catch (e) {
        alert("LLM loop failed at step " + (i + 1) + ": " + e);
        break;
      }
    }
  }
</script>

</body>

# **What’s New in This Version**

## **1. Navigation Pane**
A fixed top bar that:

- Provides orientation  
- Stays visible during scrolling  
- Makes the workspace feel like a real application  

## **2. Posted Logs Panel (Read‑Only)**
A right‑side panel that:

- Shows every iteration in a clean, readable format  
- Updates automatically whenever a new iteration is created  
- Is suitable for both humans and LLMs to read  
- Mirrors the structure of a postable iteration  
- Is strictly read‑only  

This is now a **live feed** of all reasoning steps.

# **LLM Notes**

### **LLMs can now:**
- Read the posted logs panel  
- Understand the full reasoning history  
- Produce new iterations based on the entire chain  
- Avoid repeating prior content  
- Build on previous steps  

### **LLMs should treat the logs panel as:**
- A read‑only memory buffer  
- A chronological reasoning trace  
- A source of context for the next iteration

        div.innerHTML = formatted fields
        append to logList
